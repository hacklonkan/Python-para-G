    <script>
        let pyodide;
        const editor = CodeMirror.fromTextArea(document.getElementById("codigo-base"), {
            mode: "python", theme: "dracula", lineNumbers: true, indentUnit: 4,
            extraKeys: {"Tab": (cm) => cm.replaceSelection("    ")}
        });

        // Función para insertar símbolos desde la barra
        function insertar(sym) {
            const doc = editor.getDoc();
            const cursor = doc.getCursor();
            doc.replaceRange(sym, cursor);
            editor.focus();
        }

        // Cargar código y Autoguardado
        const CODIGO_INICIAL = `limpiar_pantalla()\nagregar_titulo("App Móvil")\nagregar_boton("Púlsame", lambda e: alerta("¡Hola!"))`;
        editor.setValue(localStorage.getItem("py_code") || CODIGO_INICIAL);
        editor.on("change", () => {
            localStorage.setItem("py_code", editor.getValue());
            document.getElementById("save-status").innerText = "💾 " + new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'});
        });

        const framework = `
import js
def limpiar_pantalla(): js.document.getElementById("app-preview").innerHTML = ""
def agregar_titulo(t):
    el = js.document.createElement("h2"); el.innerHTML = t
    js.document.getElementById("app-preview").appendChild(el)
def agregar_boton(t, f):
    el = js.document.createElement("button"); el.innerHTML = t
    el.setAttribute("style", "padding:12px; background:#6272a4; color:white; border:none; border-radius:5px; width:100%; font-size:16px;")
    el.onclick = f; js.document.getElementById("app-preview").appendChild(el)
def alerta(m): js.alert(m)
`;

        async function iniciar() {
            pyodide = await loadPyodide({ stdout: (t) => { document.getElementById("consola").value += t + "\n"; } });
            document.getElementById("btn-ejecutar").disabled = false;
            document.getElementById("btn-ejecutar").innerText = "▶ EJECUTAR";
        }

        document.getElementById("btn-ejecutar").addEventListener("click", async () => {
            document.getElementById("consola").value = "";
            try { await pyodide.runPythonAsync(framework + "\n" + editor.getValue()); }
            catch (e) { document.getElementById("consola").value += "❌ " + e; }
        });

        function cambiarTema() {
            const t = document.getElementById("tema-selector").value;
            document.getElementById("main-body").className = (t === "claro" ? "claro" : "");
            editor.setOption("theme", (t === "claro" ? "default" : "dracula"));
        }

        function descargar() {
            const a = document.createElement("a");
            a.href = URL.createObjectURL(new Blob([editor.getValue()], {type:"text/plain"}));
            a.download = "mi_app.py"; a.click();
        }

        iniciar();
    </script>
